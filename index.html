    Installing the plugin:

    1. Include jQuery, mosaicPhoto script and styles 
                
<script src="//ajax.googleapis.com/ajax/libs/jquery/1.6.2/jquery.min.js"></script>
<script src="js/jquery.mosaicPhoto.js"></script>
<link rel="stylesheet" href="css/mosaicPhoto.css">
    2. create the object with arrays of images like this: 
        
var images = {
	'cat1' : [
		["/img/cat1/1.jpg", "/img/cat1/big/1.jpg"],
		 "/img/cat1/2.jpg",
		 "/img/cat1/3.jpg",
		["/img/cat1/4.jpg", "/img/cat1/big/4.jpg"],
		 "/img/cat1/5.jpg",
		 "/img/cat1/6.jpg",
		 "/img/cat1/7.jpg",
		["/img/cat1/8.jpg", "/img/cat1/big/8.jpg"],
		 "/img/cat1/9.jpg",
		 "/img/cat1/10.jpg"
	],
	'cat2' : [
		["/img/cat2/1.jpg", "http://site.com/news"],
		["/img/cat2/2.jpg", "http://site.com/about"],
		["/img/cat2/3.jpg", "http://site.com/contacts"],
		["/img/cat2/4.jpg", "http://site.com/feedback"],
		["/img/cat2/5.jpg", "http://site.com/catalog"]
	]
	};
      You can have several categories of images and each image can have a link to the full image or to some other URL.
    3. Create the container element 
      <div id="previewContainer" class="mosaicContainer"></div>
    4. Finally launch the mosaicPhoto script 
        
	jQuery(document).ready(function($){


		$("#previewContainer").mosaicPhoto({
			  images: images
			, listeners: "#menu1"
			, listenersEvents: 'click'
		});

	});
